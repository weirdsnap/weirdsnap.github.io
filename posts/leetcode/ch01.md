# 接雨水

## DP 思路

DP 左侧最大。右侧最大。 两侧最大取其小计算当前位置能接多少水。

### cpp实现

```lang=cpp
class Solution {
public:
    int trap(vector<int>& height) {
        const int length = height.size();
        // left max
        vector<int> left_max(length); 
        left_max[0] = height[0];
        for (int i = 1; i < length; i++) {
            left_max[i] = std::max(left_max[i-1], height[i]);
        }
        // right max
        vector<int> right_max(length);
        right_max[length - 1] = height[length - 1];
        for (int i = length - 2; i >= 0; i--) {
            right_max[i] = std::max(right_max[i+1], height[i]);
        }
        // result
        int result = 0;
        for (int i = 0; i < length; i++) {
            int temp = std::min(left_max[i], right_max[i]) - height[i];
            if (temp > 0) {
                result += temp;
            }
        }
        return result;
    }
};
```

## 双指针 思路

双指针  哪边小先结算那边，只要知道另一遍肯定更大那就用小的这边去结算这个位置。


### 实现

```
class Solution {
public:
    int trap(vector<int>& height) {
        const int length = height.size();
        int left = 0;
        int right = length - 1;
        int leftmax = 0;
        int rightmax = 0;
        int result = 0;
        while (left < right) {
            // 左边小结算左边
            if (height[left] < height[right]) {
                if (height[left] >= leftmax) {
                    leftmax = height[left];
                } else {
                    result += leftmax - height[left];
                }
                left++;
            } else {
                if (height[right] >= rightmax) {
                    rightmax = height[right];
                } else {
                    result += rightmax - height[right];
                }
                right--;
            }
        }
        
        return result;
    }
};
```

## 单调栈

### 实现

```
class Solution {
public:
    int trap(vector<int>& height) {
        const int length = height.size();
        vector<int> de_stack;
        int result = 0;
        int left, right, top, bottom;
        de_stack.push_back(0);  // index

        for (int i = 1; i < length; i++) {
            while (!de_stack.empty() && (height[i] >= height[de_stack.back()])) {
                bottom = height[de_stack.back()];
                de_stack.pop_back();
                if (de_stack.empty()) {
                    break;
                }
                left = height[de_stack.back()];
                right = height[i];
                top = std::min(left, right);
                result += (top - bottom) * (i - de_stack.back() - 1);
            }
            de_stack.push_back(i);
        }

      
        return result;
    }
};
```
